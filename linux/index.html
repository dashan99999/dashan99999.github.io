# 常用linux指令

# 以MB为单位显示当前文件夹下文件大小

df -h
du -sh *

# 找出当前目录下,及其子目录下大于或者小于多少M的文件

  • 找出当前目录，及其子目录中，小于1M的所有文件
find ./ -type f -size -1M
  • 找出当前目录，及其子目录中，大于1G的所有文件
find ./ -type f -size +1G

# 把当前图片压缩到500kb以内, 并覆盖当前图片

  • 安装ImageMagick
yum -y install ImageMagick
  • 压缩图片到500kb以内
/usr/bin/convert a.jpg -quality 95 -define jpeg:extent=500KB a.jpg

# 递归压缩当前文件夹及其子文件夹下的图片大小,并直接替换

find /sources -type f -name "*.jpg" -size +1M -exec convert "{}" -quality 95 -define jpeg:extent=500KB "{}" \;

/usr/bin/convert /sources/toptoon/toptoon/sbs/logo/13512/b.jpg -quality 95 -define jpeg:extent=500KB /sources/toptoon/toptoon/sbs/logo/13512/b.jpg



find /sources/toptoon/toptoon/sbs -type f -name "*.jpg" -size +1M -exec convert "{}" -quality 95 -define jpeg:extent=500KB "{}" \;
